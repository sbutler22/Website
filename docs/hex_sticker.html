                         midPoint = 12,

  reps
)

intensity <- rnorm(
  n = length(mean_vals),

  sd = rep(noise_parameter, length(mean_vals))

# --- Fit the sigmoid curve ---
sig_fit <- fitAndCategorize(dataInput, use_h0 = TRUE,)

  time,

)
